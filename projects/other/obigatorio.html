from sklearn import tree
from sklearn import model_selection
from sklearn import preprocessing

label_encoder = preprocessing.LabelEncoder()
ds= ds.apply(label_encoder.fit_transform)

#Me quedo con la variable de prediccion
Y = np.array(ds['readmitted'])

#Ajusto el tamao de los datos para test
validation_size = 0.30

#Utilizo una semilla para poder reproducir el split
seed = 2018

X=np.array(ds.drop(['readmitted'],axis=1))


#Hago un split para validar el modelo
X_train, X_validation, Y_train, Y_validation = model_selection.train_test_split(X,Y, test_size=validation_size, random_state=seed)

#Defino el KFold
kfold = model_selection.KFold(n_splits=10, random_state=seed)

#Defino el modelo
model_dtree = tree.DecisionTreeClassifier(criterion = "gini",max_depth=15)
#Obtengo los resultados del entrenamiento 
results_dtree = model_selection.cross_val_score(model_dtree, X_train,Y_train, cv=kfold, scoring='accuracy')
print("#### Resultado Training ####\n")
print("Media: %f" % (results_dtree.mean()))
print("Desvio Estandar: %f" % (results_dtree.std()))
